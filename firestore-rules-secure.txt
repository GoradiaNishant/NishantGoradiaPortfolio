// SECURE Firestore Rules - Copy these to Firebase Console

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Allow read access to portfolio data (public)
    match /portfolio/{document} {
      allow read: if true;
      allow write: if request.auth != null; // Only authenticated users can write
    }
    
    // Allow read access to projects (public)
    match /projects/{document} {
      allow read: if true;
      allow write: if request.auth != null; // Only authenticated users can write
    }
    
    // Alternative: Allow specific email addresses
    // match /{document=**} {
    //   allow read: if true;
    //   allow write: if request.auth != null && 
    //     (request.auth.token.email == "your-email@domain.com" ||
    //      request.auth.token.email_verified == true);
    // }
  }
}
